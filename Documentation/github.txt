git remote add origin [url]             // افزودن پروژه به مخزن گیت هاب

git clone [url]                         // دریافت یک کلون از پروژه

git pull origin [current branch] 		// دریافت تغییرات از محیط گیت هاب

git push origin [current branch] 		// ذخیره تغییرات در محیط گیت هاب

git remote                              // نمایش ریموت فعلی در گیت هاب

git push origin -- tags                 // ذخیره تگ ها به محیط گیت هاب

git push origin [tag name]              // ذخیره تگ مشخص به محیط گیت هاب

git checkout [tag name]                 // برگشتن به تگ مشخص
